\subsection{UC-20}
\label{subsec:UC-20}

\begin{usecase}{UC-20}{Sovrascrittura dataset corrente}

    \req{} 

    \pre{
        \item Il sistema è attivo e funzionante
        \item Il dataset corrente non è vuoto
    }

    \post{
        \item Il dataset corrente viene sostituito
        \item Il dataset sostituito non è più recuperabile
    }
    
    \actor{Utente}

    \subactors{}

    \trigger{L'utente deve caricare un dataset tra quelli archiviati nel sistema senza salvare il dataset corrente}
    
    \inc{}

    \base{}

    \scenario{
        \item L'utente conferma la sovrascrittura
        \item Il dataset corrente viene sovrascritto
    }

    \subscenario{
        \item[1.1] \textbf{L'utente non conferma la sovrascrittura del dataset corrente}
        \begin{itemize}
            \item[a.] Il dataset corrente resta invariato
        \end{itemize}
    }
\end{usecase}
