\subsection{UC-18}
\label{subsec:UC-18}

\begin{usecase}{UC-18}{Salvataggio di un nuovo dataset}

    \req{} 

    \pre{
        \item Il sistema è attivo e funzionante
        \item Il dataset corrente è temporaneo e non possiede quindi una versione salvata nel sistema
    }

    \post{
        \item Il dataset corrente viene salvato nel sistema
    }
    
    \actor{Utente}

    \subactors{}

    \trigger{Il sistema deve salvare il dataset corrente}
    
    \inc{}

    \base{}

    \scenario{
        \item Viene richiesto all'utente l'inserimento di un nome per il dataset corrente
        \item Il dataset corrente viene salvato nel sistema
    }

    \subscenario{
        \item[1.1] \textbf{Il nome del dataset è vuoto o esiste già}
        \begin{itemize}
            \item[a.] \hyperref[subsec:UC-8]{UC-8}
        \end{itemize}
        \item[2.1] \textbf{Avviene un errore durante il salvataggio}
        \begin{itemize}
            \item[a.] \hyperref[subsec:UC-11]{UC-11}
        \end{itemize}
    }
\end{usecase}
