\subsection{UC-19}
\label{subsec:UC-19}

\begin{usecase}{UC-19}{Sovrascrittura dataset caricato}

    \req{} 

    \pre{
        \item Il sistema è attivo e funzionante
        \item Il dataset caricato non è vuoto
    }

    \post{
        \item Il dataset caricato viene sostituito
        \item Il dataset sostituito non è più recuperabile
    }
    
    \actor{Utente}

    \subactors{}

    \trigger{L'utente deve caricare un dataset tra quelli archiviati nel sistema senza salvare il dataset caricato}
    
    \inc{}

    \base{}

    \scenario{
        \item L'utente conferma la sovrascrittura
        \item Il dataset caricato viene sovrascritto
    }

    \subscenario{
        \item[1.1] \textbf{L'utente non conferma la sovrascrittura del dataset caricato}
        \begin{itemize}
            \item[a.] Il dataset caricato resta invariato
        \end{itemize}
    }
\end{usecase}
